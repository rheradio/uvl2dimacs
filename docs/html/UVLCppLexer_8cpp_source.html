461void UVLCppLexer::NEWLINEAction(antlr4::RuleContext *context, size_t actionIndex) {
462 switch (actionIndex) {
463 case 8:
464 {
465 std::string newLine = getText();
466 std::string spaces = getText();
467 // Remove non-newline characters from newLine
468 newLine.erase(std::remove_if(newLine.begin(), newLine.end(),
469 [](char c) { return c != '\r' && c != '\n' && c != '\f'; }),
470 newLine.end());
471 // Remove newline characters from spaces
472 spaces.erase(std::remove_if(spaces.begin(), spaces.end(),
473 [](char c) { return c == '\r' || c == '\n' || c == '\f'; }),
474 spaces.end());
475
476 int next = _input->LA(1);
477 int nextNext = _input->LA(2);
478
479 if (opened > 0 || next == '\r' || next == '\n' || (next == '/' && nextNext == '/')) {
480 // If we're inside a list or on a blank line, ignore all indents,
481 // dedents and line breaks.
482 skip();
483 } else {
484 emitToken(commonToken(NEWLINE, newLine));
485 int indent = getIndentationCount(spaces);
486 int previous = indents.empty() ? 0 : indents.top();
487 if (indent == previous) {
488 // skip indents of the same size as the present indent-size
489 skip();
490 }
491 else if (indent > previous) {
492 indents.push(indent);
493 emitToken(commonToken(INDENT, spaces));
494 skip();
495 }
496 else {
497 // Possibly emit more than 1 DEDENT token.
498 while(!indents.empty() && indents.top() > indent) {
499 emitToken(createDedent());
500 indents.pop();
501 }
502 skip();
503 }
504 }
505 }
506 break;
507
508 default:
509 break;
510 }
