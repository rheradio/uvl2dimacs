73void DimacsWriter::write_to_stream(std::ostream& out) {
74 // Write problem line
75 out << "p cnf " << cnf_model.get_num_variables() << " " << cnf_model.get_num_clauses() << "\n";
76
77 // Write comment lines for feature variables
78 const auto& features = cnf_model.get_features();
79 for (const auto& [var_id, feature_name] : features) {
80 out << "c " << var_id << " " << feature_name << "\n";
81 }
82
83 // Write comment lines for auxiliary variables (user requested this)
84 const auto& aux_vars = cnf_model.get_auxiliary_variables();
85 for (const auto& [var_id, aux_name] : aux_vars) {
86 out << "c " << var_id << " " << aux_name << "\n";
87 }
88
89 // Write clauses
90 const auto& clauses = cnf_model.get_clauses();
91 for (const auto& clause : clauses) {
92 for (int literal : clause) {
93 out << literal << " ";
94 }
95 out << "0\n";
96 }
97}
