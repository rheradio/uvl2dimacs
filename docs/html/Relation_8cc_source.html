89Relation::Type Relation::determine_type() const {
90 size_t num_children = children.size();
91
92 // Mandatory: 1 child, [1..1]
93 if (num_children == 1 && card_min == 1 && card_max == 1) {
94 return Type::MANDATORY;
95 }
96
97 // Optional: 1 child, [0..1]
98 if (num_children == 1 && card_min == 0 && card_max == 1) {
99 return Type::OPTIONAL;
100 }
101
102 // Alternative: multiple children, [1..1]
103 if (num_children > 1 && card_min == 1 && card_max == 1) {
104 return Type::ALTERNATIVE;
105 }
106
107 // Or: multiple children, [1..N] where N is number of children
108 if (num_children > 1 && card_min == 1 && card_max == static_cast<int>(num_children)) {
109 return Type::OR;
110 }
111
112 // Everything else is a general cardinality constraint
113 return Type::CARDINALITY;
114}
