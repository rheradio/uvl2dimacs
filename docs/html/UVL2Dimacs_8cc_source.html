55class CustomErrorListener : public BaseErrorListener {
56public:
57 void syntaxError(
58 Recognizer *recognizer,
59 Token *offendingSymbol,
60 size_t line,
61 size_t charPositionInLine,
62 const std::string &msg,
63 std::exception_ptr e) override {
64
65 // Ignore tab-related warnings
66 if (msg.find("\\t") != std::string::npos) {
67 std::cerr << "Warning at line " << line << ":" << charPositionInLine
68 << " - " << msg << std::endl;
69 return;
70 }
71
72 // For other errors, throw exception
73 std::ostringstream oss;
74 oss << "The UVL has the following error that prevents reading it: Line "
75 << line << ":" << charPositionInLine << " - " << msg;
76 throw std::runtime_error(oss.str());
77 }
