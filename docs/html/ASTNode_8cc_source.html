163bool ASTNode::is_pure_boolean_tree() const {
164 // Literals are pure boolean
165 if (type == Type::LITERAL) {
166 return true;
167 }
168
169 // Non-operation types (INTEGER, FLOAT, STRING) are not pure boolean
170 if (type != Type::OPERATION) {
171 return false;
172 }
173
174 // Check if this operation is boolean
175 if (!is_boolean_operation()) {
176 return false; // Comparison or arithmetic operation
177 }
178
179 // Recursively check all children
180 for (const auto& child : children) {
181 if (!child->is_pure_boolean_tree()) {
182 return false;
183 }
184 }
185
186 return true;
187}

314int ASTNode::tseitin_transform(
315 std::vector<std::vector<int>>& clauses,
316 std::function<int(const std::string&)> get_variable,
317 std::function<int()> create_aux_var
318) const {
319 // Base case: literal
320 if (type == Type::LITERAL) {
321 return get_variable(literal_value);
322 }
323
324 // For comparison and arithmetic operations, treat them as atomic boolean propositions
325 // Create a pseudo-variable name from the string representation
326 if (!is_boolean_operation()) {
327 std::string atom_name = "_cmp_" + to_string();
328 return get_variable(atom_name);
329 }
330
331 // Handle boolean operations with Tseitin transformation
332 switch (operation) {
333 case ASTOperation::NOT: {
334 if (children.size() != 1) {
335 throw std::runtime_error("NOT operation must have exactly 1 child");
336 }
337 int child_var = children[0]->tseitin_transform(clauses, get_variable, create_aux_var);
338 int result_var = create_aux_var();
339 add_not_clauses(result_var, child_var, clauses);
340 return result_var;
341 }
342
343 case ASTOperation::AND: {
344 if (children.size() != 2) {
345 throw std::runtime_error("AND operation must have exactly 2 children");
346 }
347 int left_var = children[0]->tseitin_transform(clauses, get_variable, create_aux_var);
348 int right_var = children[1]->tseitin_transform(clauses, get_variable, create_aux_var);
349 int result_var = create_aux_var();
350 add_and_clauses(result_var, left_var, right_var, clauses);
351 return result_var;
352 }
353
354 case ASTOperation::OR: {
355 if (children.size() != 2) {
356 throw std::runtime_error("OR operation must have exactly 2 children");
357 }
358 int left_var = children[0]->tseitin_transform(clauses, get_variable, create_aux_var);
359 int right_var = children[1]->tseitin_transform(clauses, get_variable, create_aux_var);
360 int result_var = create_aux_var();
361 add_or_clauses(result_var, left_var, right_var, clauses);
362 return result_var;
363 }
364
365 case ASTOperation::IMPLIES: {
366 if (children.size() != 2) {
367 throw std::runtime_error("IMPLIES operation must have exactly 2 children");
368 }
369 int left_var = children[0]->tseitin_transform(clauses, get_variable, create_aux_var);
370 int right_var = children[1]->tseitin_transform(clauses, get_variable, create_aux_var);
371 int result_var = create_aux_var();
372 add_implies_clauses(result_var, left_var, right_var, clauses);
373 return result_var;
374 }
375
376 case ASTOperation::EQUIVALENCE: {
377 if (children.size() != 2) {
378 throw std::runtime_error("EQUIVALENCE operation must have exactly 2 children");
379 }
380 int left_var = children[0]->tseitin_transform(clauses, get_variable, create_aux_var);
381 int right_var = children[1]->tseitin_transform(clauses, get_variable, create_aux_var);
382 int result_var = create_aux_var();
383 add_equivalence_clauses(result_var, left_var, right_var, clauses);
384 return result_var;
385 }
386
387 default:
388 throw std::runtime_error("Unsupported boolean operation in Tseitin transformation");
389 }

515std::shared_ptr<ASTNode> ASTNode::to_nnf(bool negated) const {
516 // Base case: literal
517 if (type == Type::LITERAL) {
518 if (negated) {
519 // Create NOT of this literal
520 return std::make_shared<ASTNode>(ASTOperation::NOT,
521 std::make_shared<ASTNode>(literal_value));
522 } else {
523 return std::make_shared<ASTNode>(literal_value);
524 }
525 }
526
527 // Base case: numeric values (integers, floats, strings)
528 // These can't be negated, just return as-is
529 if (type == Type::INTEGER || type == Type::FLOAT || type == Type::STRING) {
530 return std::make_shared<ASTNode>(*this);
531 }
532
533 // For comparison and arithmetic operations, treat them as atomic boolean propositions
534 // We create a string representation as a pseudo-variable name
535 if (!is_boolean_operation()) {
536 std::string atom_name = "_cmp_" + to_string();
537 if (negated) {
538 return std::make_shared<ASTNode>(ASTOperation::NOT,
539 std::make_shared<ASTNode>(atom_name));
540 } else {
541 return std::make_shared<ASTNode>(atom_name);
542 }
543 }
544
545 // Handle boolean operations
546 switch (operation) {
547 case ASTOperation::NOT: {
548 // Double negation: NOT(NOT(A)) = A
549 if (children.size() != 1) {
550 throw std::runtime_error("NOT must have exactly 1 child");
551 }
552 // Flip the negation and recurse
553 return children[0]->to_nnf(!negated);
554 }
555
556 case ASTOperation::AND: {
557 if (children.size() != 2) {
558 throw std::runtime_error("AND must have exactly 2 children");
559 }
560 if (negated) {
561 // De Morgan: NOT(A AND B) = (NOT A) OR (NOT B)
562 auto left_nnf = children[0]->to_nnf(true);
563 auto right_nnf = children[1]->to_nnf(true);
564 return std::make_shared<ASTNode>(ASTOperation::OR, left_nnf, right_nnf);
565 } else {
566 // A AND B
567 auto left_nnf = children[0]->to_nnf(false);
568 auto right_nnf = children[1]->to_nnf(false);
569 return std::make_shared<ASTNode>(ASTOperation::AND, left_nnf, right_nnf);
570 }
571 }
572
573 case ASTOperation::OR: {
574 if (children.size() != 2) {
575 throw std::runtime_error("OR must have exactly 2 children");
576 }
577 if (negated) {
578 // De Morgan: NOT(A OR B) = (NOT A) AND (NOT B)
579 auto left_nnf = children[0]->to_nnf(true);
580 auto right_nnf = children[1]->to_nnf(true);
581 return std::make_shared<ASTNode>(ASTOperation::AND, left_nnf, right_nnf);
582 } else {
583 // A OR B
584 auto left_nnf = children[0]->to_nnf(false);
585 auto right_nnf = children[1]->to_nnf(false);
586 return std::make_shared<ASTNode>(ASTOperation::OR, left_nnf, right_nnf);
587 }
588 }
589
590 case ASTOperation::IMPLIES: {
591 if (children.size() != 2) {
592 throw std::runtime_error("IMPLIES must have exactly 2 children");
593 }
594 // A IMPLIES B = NOT A OR B
595 if (negated) {
596 // NOT(A IMPLIES B) = NOT(NOT A OR B) = A AND NOT B
597 auto left_nnf = children[0]->to_nnf(false);
598 auto right_nnf = children[1]->to_nnf(true);
599 return std::make_shared<ASTNode>(ASTOperation::AND, left_nnf, right_nnf);
600 } else {
601 // A IMPLIES B = NOT A OR B
602 auto left_nnf = children[0]->to_nnf(true);
603 auto right_nnf = children[1]->to_nnf(false);
604 return std::make_shared<ASTNode>(ASTOperation::OR, left_nnf, right_nnf);
605 }
606 }
607
608 case ASTOperation::EQUIVALENCE: {
609 if (children.size() != 2) {
610 throw std::runtime_error("EQUIVALENCE must have exactly 2 children");
611 }
612 // A <=> B = (A AND B) OR (NOT A AND NOT B)
613 if (negated) {
614 // NOT(A <=> B) = (A AND NOT B) OR (NOT A AND B)
615 auto left_pos = children[0]->to_nnf(false);
616 auto left_neg = children[0]->to_nnf(true);
617 auto right_pos = children[1]->to_nnf(false);
618 auto right_neg = children[1]->to_nnf(true);
619 auto and1 = std::make_shared<ASTNode>(ASTOperation::AND, left_pos, right_neg);
620 auto and2 = std::make_shared<ASTNode>(ASTOperation::AND, left_neg, right_pos);
621 return std::make_shared<ASTNode>(ASTOperation::OR, and1, and2);
622 } else {
623 // A <=> B = (A AND B) OR (NOT A AND NOT B)
624 auto left_pos = children[0]->to_nnf(false);
625 auto left_neg = children[0]->to_nnf(true);
626 auto right_pos = children[1]->to_nnf(false);
627 auto right_neg = children[1]->to_nnf(true);
628 auto and1 = std::make_shared<ASTNode>(ASTOperation::AND, left_pos, right_pos);
629 auto and2 = std::make_shared<ASTNode>(ASTOperation::AND, left_neg, right_neg);
630 return std::make_shared<ASTNode>(ASTOperation::OR, and1, and2);
631 }
632 }
633
634 default:
635 throw std::runtime_error("Unsupported operation in NNF conversion: " + to_string());
636 }
637}

660std::vector<std::vector<int>> ASTNode::to_cnf_direct(
661 std::function<int(const std::string&)> get_variable
662) const {
663 // Base case: literal
664 if (type == Type::LITERAL) {
665 int var = get_variable(literal_value);
666 return {{var}};
667 }
668
669 // Handle NOT of literal (from NNF conversion)
670 if (operation == ASTOperation::NOT) {
671 if (children.size() != 1 || children[0]->get_type() != Type::LITERAL) {
672 throw std::runtime_error("In NNF, NOT should only apply to literals");
673 }
674 int var = get_variable(children[0]->get_literal());
675 return {{-var}};
676 }
677
678 // Handle AND: concatenate clauses from both children
679 if (operation == ASTOperation::AND) {
680 if (children.size() != 2) {
681 throw std::runtime_error("AND must have exactly 2 children");
682 }
683 auto left_clauses = children[0]->to_cnf_direct(get_variable);
684 auto right_clauses = children[1]->to_cnf_direct(get_variable);
685
686 // Concatenate clauses
687 std::vector<std::vector<int>> result = left_clauses;
688 result.insert(result.end(), right_clauses.begin(), right_clauses.end());
689 return result;
690 }
691
692 // Handle OR: distribute over AND
693 if (operation == ASTOperation::OR) {
694 if (children.size() != 2) {
695 throw std::runtime_error("OR must have exactly 2 children");
696 }
697 auto left_clauses = children[0]->to_cnf_direct(get_variable);
698 auto right_clauses = children[1]->to_cnf_direct(get_variable);
699
700 // Distribute OR over AND
701 return distribute_or(left_clauses, right_clauses);
702 }
703
704 throw std::runtime_error("Unexpected operation in CNF conversion: " + to_string());
705}

728std::vector<std::vector<int>> ASTNode::distribute_or(
729 const std::vector<std::vector<int>>& left_clauses,
730 const std::vector<std::vector<int>>& right_clauses
731) const {
732 std::vector<std::vector<int>> result;
733
734 // For each clause from left and each clause from right, merge them
735 for (const auto& left_clause : left_clauses) {
736 for (const auto& right_clause : right_clauses) {
737 // Merge the two clauses (union of literals)
738 std::vector<int> merged = left_clause;
739 merged.insert(merged.end(), right_clause.begin(), right_clause.end());
740 result.push_back(merged);
741 }
742 }
743
744 return result;
