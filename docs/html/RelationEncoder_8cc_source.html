83void RelationEncoder::encode_mandatory(std::shared_ptr<Relation> relation) {
84 // Mandatory: parent <=> child
85 // Clauses: (-parent OR child) AND (-child OR parent)
86
87 auto parent = relation->get_parent();
88 const auto& children = relation->get_children();
89
90 if (children.size() != 1) {
91 throw std::runtime_error("Mandatory relation must have exactly 1 child");
92 }
93
94 int parent_var = cnf_model.get_variable(parent->get_name());
95 int child_var = cnf_model.get_variable(children[0]->get_name());
96
97 // -parent OR child
98 cnf_model.add_clause({-parent_var, child_var});
99
100 // -child OR parent
101 cnf_model.add_clause({-child_var, parent_var});
102}

148void RelationEncoder::encode_or(std::shared_ptr<Relation> relation) {
149 // Or: parent => (child1 OR child2 OR ... OR childN)
150 // AND each child => parent
151 //
152 // Clauses:
153 // 1. (-parent OR child1 OR child2 OR ... OR childN)
154 // 2. For each child i: (-child_i OR parent)
155
156 auto parent = relation->get_parent();
157 const auto& children = relation->get_children();
158
159 if (children.empty()) {
160 throw std::runtime_error("Or relation must have at least 1 child");
161 }
162
163 int parent_var = cnf_model.get_variable(parent->get_name());
164
165 // Clause 1: -parent OR child1 OR child2 OR ... OR childN
166 std::vector<int> or_clause = {-parent_var};
167 for (const auto& child : children) {
168 or_clause.push_back(cnf_model.get_variable(child->get_name()));
169 }
170 cnf_model.add_clause(or_clause);
171
172 // Clause 2: For each child, -child OR parent
173 for (const auto& child : children) {
174 int child_var = cnf_model.get_variable(child->get_name());
175 cnf_model.add_clause({-child_var, parent_var});
176 }

196void RelationEncoder::encode_alternative(std::shared_ptr<Relation> relation) {
197 // Alternative: parent => (exactly one of children)
198 //
199 // Clauses:
200 // 1. (-parent OR child1 OR child2 OR ... OR childN) [at least one]
201 // 2. For each pair (i,j): (-child_i OR -child_j) [at most one]
202 // 3. For each child i: (-child_i OR parent) [children => parent]
203
204 auto parent = relation->get_parent();
205 const auto& children = relation->get_children();
206
207 if (children.size() < 2) {
208 throw std::runtime_error("Alternative relation must have at least 2 children");
209 }
210
211 int parent_var = cnf_model.get_variable(parent->get_name());
212
213 // Clause 1: -parent OR child1 OR child2 OR ... OR childN
214 std::vector<int> or_clause = {-parent_var};
215 for (const auto& child : children) {
216 or_clause.push_back(cnf_model.get_variable(child->get_name()));
217 }
218 cnf_model.add_clause(or_clause);
219
220 // Clause 2: For each pair (i,j), -child_i OR -child_j
221 for (size_t i = 0; i < children.size(); ++i) {
222 for (size_t j = i + 1; j < children.size(); ++j) {
223 int child_i_var = cnf_model.get_variable(children[i]->get_name());
224 int child_j_var = cnf_model.get_variable(children[j]->get_name());
225 cnf_model.add_clause({-child_i_var, -child_j_var});
226 }
227 }
228
229 // Clause 3: For each child, -child OR parent
230 for (const auto& child : children) {
231 int child_var = cnf_model.get_variable(child->get_name());
232 cnf_model.add_clause({-child_var, parent_var});
233 }
234}

262void RelationEncoder::encode_cardinality(std::shared_ptr<Relation> relation) {
263 // Cardinality: parent => (select min..max children)
264 // Uses enumeration approach from Python implementation
265 //
266 // For each invalid count (< min or > max):
267 // For each combination of that count:
268 // (-parent OR NOT(exactly this combination))
269 //
270 // Special case for valid counts with parent selected:
271 // For each valid count > 0:
272 // For each combination of that count:
273 // (parent OR NOT(exactly this combination))
274
275 auto parent = relation->get_parent();
276 const auto& children = relation->get_children();
277 int card_min = relation->get_card_min();
278 int card_max = relation->get_card_max();
279 int num_children = children.size();
280
281 int parent_var = cnf_model.get_variable(parent->get_name());
282
283 // For each possible count of selected children
284 for (int count = 0; count <= num_children; ++count) {
285 bool is_invalid = (count < card_min || count > card_max);
286
287 // Generate all combinations of 'count' children
288 auto combinations = generate_combinations(num_children, count);
289
290 for (const auto& combo : combinations) {
291 std::vector<int> clause;
292
293 if (is_invalid) {
294 // Invalid count: -parent OR NOT(exactly this combination)
295 clause.push_back(-parent_var);
296 } else if (count > 0) {
297 // Valid count > 0: parent OR NOT(exactly this combination)
298 clause.push_back(parent_var);
299 } else {
300 // count == 0 and is valid: this handles the case where min=0
301 // -parent OR NOT(exactly this combination)
302 clause.push_back(-parent_var);
303 }
304
305 // Add literals for "NOT(exactly this combination)"
306 // This means: at least one child in combo is false OR at least one child not in combo is true
307 for (int i = 0; i < num_children; ++i) {
308 int child_var = cnf_model.get_variable(children[i]->get_name());
309 bool in_combo = std::find(combo.begin(), combo.end(), i) != combo.end();
310
311 if (in_combo) {
312 // This child should be false (negated)
313 clause.push_back(-child_var);
314 } else {
315 // This child should be true (positive)
316 clause.push_back(child_var);
317 }
318 }
319
320 cnf_model.add_clause(clause);
321 }
322 }
323
324 // Add clauses: each child => parent
325 for (const auto& child : children) {
326 int child_var = cnf_model.get_variable(child->get_name());
327 cnf_model.add_clause({-child_var, parent_var});
328 }
329}

350std::vector<std::vector<int>> RelationEncoder::generate_combinations(int n, int k) {
351 std::vector<std::vector<int>> result;
352
353 if (k > n || k < 0) {
354 return result;
355 }
356
357 if (k == 0) {
358 result.push_back({});
359 return result;
360 }
361
362 // Generate combinations using backtracking
363 std::vector<int> current;
364 std::function<void(int, int)> backtrack = [&](int start, int depth) {
365 if (depth == k) {
366 result.push_back(current);
367 return;
368 }
369
370 for (int i = start; i < n; ++i) {
371 current.push_back(i);
372 backtrack(i + 1, depth + 1);
373 current.pop_back();
374 }
375 };
376
377 backtrack(0, 0);
378 return result;
