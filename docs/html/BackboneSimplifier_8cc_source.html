115bool BackboneSimplifier::parse_backbone_output(const std::string& output, int num_vars) {
116 // Remove ANSI escape codes
117 std::regex ansi_regex(R"(\x1b\[[0-9;]*m)");
118 std::string clean_output = std::regex_replace(output, ansi_regex, "");
119
120 // Find the backbone line (starts with 'v ')
121 std::istringstream iss(clean_output);
122 std::string line;
123 bool found_backbone = false;
124
125 while (std::getline(iss, line)) {
126 if (line.length() >= 2 && line[0] == 'v' && line[1] == ' ') {
127 found_backbone = true;
128
129 // Parse literals
130 std::istringstream literal_stream(line.substr(2));
131 int literal;
132
133 while (literal_stream >> literal) {
134 if (literal == 0) break;
135
136 int var_id = std::abs(literal);
137 if (var_id > 0 && var_id <= num_vars) {
138 if (literal > 0) {
139 backbone_[var_id] = BackboneState::TRUE;
140 } else {
141 backbone_[var_id] = BackboneState::FALSE;
142 }
143 backbone_size_++;
144 }
145 }
146 break;
147 }
148 }
149
150 if (!found_backbone) {
151 // Empty backbone is valid (no backbone literals)
152 if (verbose_) {
153 std::cout << "No backbone found (formula has flexible variables)" << std::endl;
154 }
155 return true;
156 }
157
158 return true;

269bool BackboneSimplifier::process_clause(std::vector<int>& clause) {
270 if (clause.empty() || clause.back() != 0) {
271 return true; // Keep malformed clauses as-is
272 }
273
274 std::vector<int> literals_to_remove;
275 bool clause_shortened = false;
276
277 // Check each literal (except the terminating 0)
278 for (size_t i = 0; i < clause.size() - 1; ++i) {
279 int literal = clause[i];
280 int var_id = std::abs(literal);
281
282 if (var_id <= 0 || var_id >= (int)backbone_.size()) {
283 continue; // Invalid variable ID
284 }
285
286 if (literal > 0) {
287 // Positive literal
288 if (backbone_[var_id] == BackboneState::TRUE) {
289 // Clause is satisfied, remove it
290 return false;
291 } else if (backbone_[var_id] == BackboneState::FALSE) {
292 // Literal is false, remove it from clause
293 literals_to_remove.push_back(i);
294 if (!clause_shortened) {
295 shortened_clauses_++;
296 clause_shortened = true;
297 }
298 }
299 } else {
300 // Negative literal
301 if (backbone_[var_id] == BackboneState::FALSE) {
302 // Clause is satisfied, remove it
303 return false;
304 } else if (backbone_[var_id] == BackboneState::TRUE) {
305 // Literal is false, remove it from clause
306 literals_to_remove.push_back(i);
307 if (!clause_shortened) {
308 shortened_clauses_++;
309 clause_shortened = true;
310 }
311 }
312 }
313 }
314
315 // Remove literals in reverse order to maintain indices
316 for (auto it = literals_to_remove.rbegin(); it != literals_to_remove.rend(); ++it) {
317 clause.erase(clause.begin() + *it);
318 }
319
320 // Keep the clause
321 return true;
322}
